for ((index,e) in array.withIndex()){
    println("下标=$index----元素=$e")
}

遍历一个数组/列表，只取出下标:

val array = arrayOf("a", "b", "c")
for (index in array.indices){
    println("index=$index")//输出0，1，2
}
